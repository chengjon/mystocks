29def find_env_file(default_path="mystocks/.env"): 

30 """ 

31 智能查找环境变量文件，支持多种工作目录 

32 

33 Args: 

34 default_path (str): 默认相对路径 

35 

36 Returns: 

37 str: 找到的环境文件绝对路径 

38 

39 Raises: 

40 FileNotFoundError: 如果所有路径都找不到文件 

41 """ 

42 # 获取脚本所在目录 

43 script_dir = os.path.dirname(os.path.abspath(__file__)) 

44 

45 # 定义多个可能的路径（按优先级排序） 

46 possible_paths = [ 

47 # 1. 当前工作目录的相对路径 

48 default_path, 

49 # 2. 当前目录的 .env 文件 

50 ".env", 

51 # 3. 上级目录的 mystocks/.env 

52 "../mystocks/.env", 

53 # 4. 从脚本目录向上找到项目根目录 

54 os.path.join(script_dir, "../../../mystocks/.env"), 

55 os.path.join(script_dir, "../../.env"), 

56 os.path.join(script_dir, "../.env"), 

57 # 5. 固定的已知路径 

58 r"D:\MyData\GITHUB\mystocks\.env", 

59 # 6. 脚本目录的兄弟目录 

60 os.path.join(os.path.dirname(script_dir), ".env"), 

61 ] 

62 

63 logger.debug("🔍 开始智能搜索环境文件，默认路径: %s", default_path) 

64 logger.debug("📁 脚本所在目录: %s", script_dir) 

65 logger.debug("📂 当前工作目录: %s", os.getcwd()) 

66 

67 for i, path in enumerate(possible_paths, 1): 

68 try: 

69 # 转换为绝对路径 

70 abs_path = os.path.abspath(path) 

71 logger.debug("📋 [%s/%s] 检查路径: %s", i, len(possible_paths), abs_path) 

72 

73 if os.path.exists(abs_path): 

74 logger.success(f"✅ 找到环境文件: {abs_path}") 

75 return abs_path 

76 else: 

77 logger.debug("❌ 路径不存在: %s", abs_path) 

78 

79 except Exception as e: 

80 logger.debug("⚠️ 检查路径时出错: %s - %s", path, str(e)) 

81 continue 

82 

83 # 如果所有路径都找不到，抛出详细错误 

84 error_msg = f""" 

85环境变量文件未找到！已尝试以下路径： 

86{"".join([f" {i}. {os.path.abspath(path)}\n" for i, path in enumerate(possible_paths, 1)])} 

87请确保： 

881. .env 文件存在于正确位置 

892. 当前工作目录正确 (当前: {os.getcwd()}) 

903. 文件路径权限正确 

91""" 

92 

93 logger.error(error_msg) 

94 raise FileNotFoundError(error_msg)

97def load_env_config(env_file=None): 

98 """从环境变量文件加载配置""" 

99 # 如果没有指定路径，使用智能搜索 

100 if env_file is None: 

101 env_file = find_env_file() 

102 else: 

103 # 如果指定了路径，先检查是否存在，不存在则使用智能搜索 

104 if not os.path.exists(env_file): 

105 logger.warning("⚠️ 指定的环境文件不存在: %s，尝试智能搜索...", env_file) 

106 env_file = find_env_file() 

107 else: 

108 env_file = os.path.abspath(env_file) 

109 

110 logger.info("🔍 开始加载环境配置文件: %s", env_file) 

111 config = {} 

112 start_time = time.time() 

113 

114 try: 

115 logger.success(f"✓ 环境文件存在: {env_file}") 

116 

117 # 读取文件内容 

118 with open(env_file, "r", encoding="utf-8") as f: 

119 lines = f.readlines() 

120 logger.info("📄 读取到 %s 行配置", len(lines)) 

121 

122 for line_num, line in enumerate(lines, 1): 

123 line = line.strip() 

124 # 跳过注释和空行 

125 if not line or line.startswith("#"): 

126 continue 

127 

128 # 解析键值对 

129 if "=" in line: 

130 key, value = line.split("=", 1) 

131 config[key.strip()] = value.strip() 

132 logger.debug("第%s行: 加载配置 %s", line_num, key.strip()) 

133 

134 # 验证必要的配置项 

135 required_keys = ["MYSQL_HOST", "MYSQL_USER", "MYSQL_PASSWORD", "MYSQL_PORT"] 

136 missing_keys = [key for key in required_keys if key not in config] 

137 

138 if missing_keys: 

139 raise ValueError(f"环境变量文件缺少必要配置: {', '.join(missing_keys)}") 

140 

141 # 构建数据库配置 

142 db_config = { 

143 "user": config["MYSQL_USER"], 

144 "password": config["MYSQL_PASSWORD"], 

145 "host": config["MYSQL_HOST"], 

146 "port": int(config["MYSQL_PORT"]), 

147 "database": "mysql", # 初始连接使用的数据库 

148 "charset": "utf8mb4", 

149 "collation": "utf8mb4_unicode_ci", 

150 } 

151 

152 load_time = time.time() - start_time 

153 logger.success(f"✓ 环境配置加载成功! 耗时: {load_time:.3f}s") 

154 logger.info("🔗 数据库连接信息: {config['MYSQL_USER']}@{config['MYSQL_HOST']}:{config['MYSQL_PORT']}") 

155 

156 return db_config 

157 

158 except Exception as e: 

159 load_time = time.time() - start_time 

160 logger.error("❌ 加载配置失败 (耗时: %ss): %s", load_time, str(e)) 

161 raise

255def create_database_and_tables(drop_existing=False): 

256 """创建数据库和表结构""" 

257 logger.info("🚀 开始创建数据库和表结构 (drop_existing=%s)", drop_existing) 

258 start_time = time.time() 

259 

260 try: 

261 # 从 env 文件加载配置 

262 db_config = load_env_config() 

263 

264 # 创建数据库连接字符串 

265 connection_str = ( 

266 f"mysql+pymysql://{db_config['user']}:{db_config['password']}@" 

267 f"{db_config['host']}:{db_config['port']}/{db_config['database']}?" 

268 f"charset={db_config['charset']}" 

269 ) 

270 

271 logger.info("🔗 连接数据库: {db_config['user']}@{db_config['host']}:{db_config['port']}") 

272 

273 # 建立数据库连接 

274 engine = sqlalchemy.create_engine(connection_str) 

275 with engine.connect() as connection: 

276 # 确保自动提交模式开启 

277 connection = connection.execution_options(autocommit=True) 

278 logger.success("✓ 数据库连接成功") 

279 

280 # 获取 SQL 命令 

281 sql_commands = get_sql_commands( 

282 drop_existing=drop_existing, 

283 charset=db_config["charset"], 

284 collation=db_config["collation"], 

285 ).split(";") 

286 

287 # 统计信息 

288 total_commands = len([cmd for cmd in sql_commands if cmd.strip()]) 

289 executed_commands = 0 

290 failed_commands = 0 

291 

292 logger.info("📄 将执行 %s 条 SQL 命令", total_commands) 

293 

294 # 执行 SQL 命令 

295 for i, cmd in enumerate(sql_commands, 1): 

296 cmd = cmd.strip() 

297 if cmd: # 跳过空命令 

298 cmd_start_time = time.time() 

299 try: 

300 # 判断命令类型 

301 if "CREATE DATABASE" in cmd: 

302 logger.info("📁 [%s/%s] 创建数据库: db_monitor", i, total_commands) 

303 elif "USE db_monitor" in cmd: 

304 logger.info("🔄 [%s/%s] 切换到数据库: db_monitor", i, total_commands) 

305 elif "CREATE TABLE" in cmd: 

306 table_name = extract_table_name(cmd) 

307 logger.info("📊 [%s/%s] 创建表: %s", i, total_commands, table_name) 

308 elif "DROP TABLE" in cmd: 

309 logger.warning("🗑️ [%s/%s] 删除表", i, total_commands) 

310 else: 

311 logger.debug("📋 [%s/%s] 执行 SQL: %s...", i, total_commands, cmd[:100]) 

312 

313 connection.execute(text(cmd)) 

314 cmd_time = time.time() - cmd_start_time 

315 executed_commands += 1 

316 

317 if cmd_time > 0.1: # 只记录较慢的命令 

318 logger.debug("⏱️ 命令执行时间: %ss", cmd_time) 

319 

320 except Exception as cmd_error: 

321 cmd_time = time.time() - cmd_start_time 

322 failed_commands += 1 

323 logger.error( 

324 "❌ [%s/%s] SQL执行失败 (耗时: %ss): %s", i, total_commands, cmd_time, str(cmd_error) 

325 ) 

326 logger.debug("失败的SQL: %s...", cmd[:200]) 

327 

328 total_time = time.time() - start_time 

329 

330 # 输出成功统计 

331 logger.success("✓ 数据库初始化完成!") 

332 logger.info("📊 执行统计: 成功 %s / 失败 %s / 总计 %s", executed_commands, failed_commands, total_commands) 

333 logger.info("⏱️ 总执行时间: %ss", total_time) 

334 

335 # 输出创建的资源汇总 

336 logger.info("📦 创建的资源汇总:") 

337 logger.info(" • 数据库: db_monitor") 

338 logger.info(" • 表结构:") 

339 tables = [ 

340 "table_creation_log - 表创建日志表", 

341 "column_definition_log - 列定义日志表", 

342 "table_operation_log - 表操作日志表", 

343 "table_validation_log - 表结构验证日志表", 

344 ] 

345 for table in tables: 

346 logger.info(" ▫ %s", table) 

347 

348 return True 

349 

350 except SQLAlchemyError as e: 

351 total_time = time.time() - start_time 

352 logger.error("❌ 执行 SQL 时发生错误 (耗时: %ss): %s", total_time, str(e)) 

353 return False 

354 except Exception as e: 

355 total_time = time.time() - start_time 

356 logger.error("❌ 发生意外错误 (耗时: %ss): %s", total_time, str(e)) 

357 return False
