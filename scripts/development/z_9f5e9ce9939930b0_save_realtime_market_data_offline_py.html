107 """获取实时市场数据"""

211 def save_to_json(self, data: pd.DataFrame, market_symbol: str = None) -> str: 

212 """保存数据到JSON文件""" 

213 if not self.config["export_json"]: 

214 return None 

215 

216 try: 

217 symbol = market_symbol or self.config["market_symbol"] 

218 timestamp = datetime.now().strftime("%Y%m%d_%H%M%S") 

219 

220 # 生成文件名 

221 filename = f"realtime_market_{symbol}_{timestamp}.json" 

222 filepath = os.path.join(self.config["backup_dir"], filename) 

223 

224 # 转换日期时间为字符串 

225 data_copy = data.copy() 

226 for col in data_copy.columns: 

227 if data_copy[col].dtype == "datetime64[ns]": 

228 data_copy[col] = data_copy[col].dt.strftime("%Y-%m-%d %H:%M:%S") 

229 

230 # 保存JSON 

231 data_dict = data_copy.to_dict(orient="records") 

232 with open(filepath, "w", encoding="utf-8") as f: 

233 json.dump(data_dict, f, ensure_ascii=False, indent=2) 

234 

235 self.logger.info("✅ 数据已保存到JSON: %s", filepath) 

236 return filepath

238 except Exception as e:

305 def run(self, market_symbol: str = None, force_update: bool = False) -> bool: 

306 """运行完整流程""" 

307 try: 

308 symbol = market_symbol or self.config["market_symbol"] 

309 

310 self.logger.info("=" * 60) 

311 self.logger.info("🚀 离线版沪深市场A股实时数据保存系统") 

312 self.logger.info("=" * 60) 

313 self.logger.info("📊 目标市场: %s", symbol) 

314 self.logger.info("🗂️ 备份目录: %s", self.config["backup_dir"]) 

315 self.logger.info("🔄 强制更新: %s", "是" if force_update else "否") 

316 self.logger.info("=" * 60) 

317 

318 # 1. 检查依赖 

319 if not self.check_dependencies(): 

320 return False 

321 

322 # 2. 获取数据（支持重试） 

323 data = None 

324 for attempt in range(self.config["max_retry_attempts"]): 

325 self.logger.info("📡 第%s次获取数据...", attempt + 1) 

326 data = self.get_realtime_market_data(symbol) 

327 if data is not None: 

328 break 

329 self.logger.warning("⚠️ 第%s次获取失败", attempt + 1) 

330 

331 if data is None: 

332 self.logger.error("💥 多次重试后仍无法获取数据") 

333 return False 

334 

335 # 3. 保存数据到多种格式 

336 saved_files = [] 

337 

338 # CSV文件（主要格式） 

339 csv_file = self.save_to_csv(data, symbol) 

340 saved_files.append(csv_file) 

341 

342 # JSON文件（可选） 

343 json_file = self.save_to_json(data, symbol) 

344 saved_files.append(json_file) 

345 

346 # Excel文件（可选） 

347 excel_file = self.save_to_excel(data, symbol) 

348 saved_files.append(excel_file) 

349 

350 # 4. 生成汇总报告 

351 self.create_summary_report(data, saved_files) 

352 

353 # 5. 显示结果 

354 success_count = len([f for f in saved_files if f is not None]) 

355 

356 self.logger.info("=" * 60) 

357 self.logger.info("🎉 数据保存完成！") 

358 self.logger.info("📊 数据记录数: %s", len(data)) 

359 self.logger.info("💾 保存文件数: %s", success_count) 

360 self.logger.info("📁 保存位置: %s", os.path.abspath(self.config["backup_dir"])) 

361 

362 # 显示文件列表 

363 for file_path in saved_files: 

364 if file_path: 

365 file_size = os.path.getsize(file_path) 

366 self.logger.info(" ✅ %s (%s 字节)", os.path.basename(file_path), file_size) 

367 

368 self.logger.info("=" * 60) 

369 

370 return True 

371 

372 except Exception as e: 

373 self.logger.error("💥 程序执行失败: %s", e) 

374 return False
