402async def _try_get_cached_dashboard( 

403 cache_manager: CacheManager, 

404 user_id: int, 

405 trade_date: Optional[date], 

406) -> tuple[Optional[Dict[str, Any]], bool]: 

407 """ 

408 尝试从三级缓存获取仪表盘数据 

409 

410 Args: 

411 cache_manager: 异步缓存管理器实例 

412 user_id: 用户ID 

413 trade_date: 交易日期 

414 

415 Returns: 

416 (缓存数据, 缓存是否命中) 

417 如果命中缓存，返回 (数据, True) 

418 如果未命中缓存，返回 (None, False) 

419 """ 

420 try: 

421 cache_key = _generate_cache_key(user_id, trade_date) 

422 cached_data = await cache_manager.fetch_from_cache( 

423 symbol=f"user_{user_id}", 

424 data_type="dashboard", 

425 timeframe="1d", 

426 ) 

427 

428 if cached_data and isinstance(cached_data, dict): 

429 logger.info(f"✅ 三级缓存命中: {cache_key}") 

430 return cached_data, True 

431 else: 

432 logger.debug(f"⚠️ 三级缓存未命中: {cache_key}") 

433 return None, False 

434 

435 except Exception as e: 

436 logger.warning(f"三级缓存读取失败，将继续获取新数据: {str(e)}") 

437 return None, False

440async def _cache_dashboard_data( 

441 cache_manager: CacheManager, 

442 user_id: int, 

443 trade_date: Optional[date], 

444 dashboard_data: Dict[str, Any], 

445 ttl_hours: int = 24, 

446) -> bool: 

447 """ 

448 将仪表盘数据写入三级缓存 

449 

450 Args: 

451 cache_manager: 异步缓存管理器实例 

452 user_id: 用户ID 

453 trade_date: 交易日期 

454 dashboard_data: 要缓存的仪表盘数据 

455 ttl_hours: 缓存生存时间（小时） 

456 

457 Returns: 

458 True 如果缓存成功，False 否则 

459 """ 

460 try: 

461 cache_key = _generate_cache_key(user_id, trade_date) 

462 

463 # 为缓存数据添加元数据 

464 cache_entry = { 

465 "dashboard_data": dashboard_data, 

466 "user_id": user_id, 

467 "trade_date": (trade_date or date.today()).isoformat(), 

468 "cached_at": datetime.now().isoformat(), 

469 "ttl_hours": ttl_hours, 

470 } 

471 

472 success = await cache_manager.write_to_cache( 

473 symbol=f"user_{user_id}", 

474 data_type="dashboard", 

475 timeframe="1d", 

476 data=cache_entry, 

477 ttl_days=(ttl_hours + 23) // 24, # 四舍五入到天 

478 timestamp=datetime.now(), 

479 ) 

480 

481 # Ensure bool return type 

482 success_bool: bool = bool(success) 

483 if success_bool: 

484 logger.info(f"✅ 三级缓存写入成功: {cache_key}") 

485 else: 

486 logger.warning(f"⚠️ 三级缓存写入失败: {cache_key}") 

487 

488 return success_bool 

489 

490 except Exception as e: 

491 logger.warning(f"三级缓存写入异常: {str(e)}") 

492 return False

505) 

506async def get_dashboard_summary( 

507 user_id: int = Query(..., description="用户ID", ge=1), 

508 trade_date: Optional[date] = Query(None, description="交易日期，默认为今天"), 

509 include_market: bool = Query(True, description="是否包含市场概览"), 

510 include_watchlist: bool = Query(True, description="是否包含自选股"), 

511 include_portfolio: bool = Query(True, description="是否包含持仓"), 

512 include_alerts: bool = Query(True, description="是否包含风险预警"), 

513 bypass_cache: bool = Query(False, description="是否跳过缓存直接获取新数据"), 

514): 

515 """ 

516 获取仪表盘汇总数据 

517 

518 **参数说明**: 

519 - user_id: 用户ID（必须） 

520 - trade_date: 交易日期（可选，默认今天） 

521 - include_market: 是否包含市场概览（默认true） 

522 - include_watchlist: 是否包含自选股（默认true） 

523 - include_portfolio: 是否包含持仓（默认true） 

524 - include_alerts: 是否包含风险预警（默认true） 

525 - bypass_cache: 是否跳过缓存（默认false） 

526 

527 **返回数据**: 

528 - 市场概览: 指数数据、涨跌家数、榜单等 

529 - 自选股: 自选股列表、平均涨跌幅等 

530 - 持仓: 持仓列表、总盈亏等 

531 - 风险预警: 预警列表、未读数量等 

532 """ 

533 cache_manager = await get_cache_manager() 

534 cache_hit = False 

535 

536 try: 

537 # 第1阶段：尝试从三级缓存获取数据 

538 if not bypass_cache: 

539 cached_entry, cache_hit = await _try_get_cached_dashboard( 

540 cache_manager, 

541 user_id, 

542 trade_date, 

543 ) 

544 if cache_hit and cached_entry: 

545 logger.info(f"三级缓存命中: user_id={user_id}, trade_date={trade_date}") 

546 raw_dashboard = cached_entry.get("dashboard_data", {}) 

547 else: 

548 raw_dashboard = None 

549 else: 

550 logger.info(f"跳过三级缓存获取仪表盘数据: user_id={user_id}") 

551 raw_dashboard = None 

552 

553 # 第2阶段：如果缓存未命中，从数据源获取新数据 

554 if not cache_hit or raw_dashboard is None: 

555 logger.info(f"从数据源获取用户{user_id}的仪表盘数据") 

556 

557 # 使用数据源工厂 

558 from app.services.data_source_factory import get_data_source_factory 

559 

560 factory = await get_data_source_factory() 

561 

562 # 构建请求参数 

563 params = { 

564 "user_id": user_id, 

565 "trade_date": trade_date.isoformat() if trade_date else None, 

566 "include_market": include_market, 

567 "include_watchlist": include_watchlist, 

568 "include_portfolio": include_portfolio, 

569 "include_alerts": include_alerts, 

570 } 

571 

572 # 调用数据源工厂获取dashboard/summary数据 

573 raw_dashboard = await factory.get_data("dashboard", "summary", params) 

574 

575 # 第3阶段：将新数据写入三级缓存 

576 await _cache_dashboard_data( 

577 cache_manager, 

578 user_id, 

579 trade_date, 

580 raw_dashboard, 

581 ttl_hours=24, # 缓存24小时 

582 ) 

583 

584 # 第4阶段：构建响应数据 

585 response = DashboardResponse( 

586 user_id=user_id, 

587 trade_date=trade_date or date.today(), 

588 generated_at=datetime.now(), 

589 data_source=raw_dashboard.get("data_source", "data_source_factory"), 

590 cache_hit=cache_hit, # ✅ 实现缓存机制：根据实际缓存命中状态更新 

591 ) 

592 

593 # 根据参数选择性包含各模块数据 

594 if include_market and "market_overview" in raw_dashboard: 

595 response.market_overview = build_market_overview(raw_dashboard["market_overview"]) 

596 

597 if include_watchlist and "watchlist" in raw_dashboard: 

598 response.watchlist = build_watchlist_summary(raw_dashboard["watchlist"]) 

599 

600 if include_portfolio and "portfolio" in raw_dashboard: 

601 response.portfolio = build_portfolio_summary(raw_dashboard["portfolio"]) 

602 

603 if include_alerts and "risk_alerts" in raw_dashboard: 

604 response.risk_alerts = build_risk_alert_summary(raw_dashboard["risk_alerts"]) 

605 

606 # 记录缓存统计 

607 cache_stats = cache_manager.get_cache_stats() 

608 logger.info( 

609 f"仪表盘数据获取成功: user_id={user_id}, cache_hit={cache_hit}, " 

610 f"hit_rate={cache_stats.get('hit_rate', 'N/A')}" 

611 ) 

612 return response 

613 

614 except ValueError as e: 

615 logger.error(f"参数验证失败: {str(e)}")
