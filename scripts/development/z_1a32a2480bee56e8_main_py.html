57# SECURITY FIX 1.2: CSRF Token管理 

58class CSRFTokenManager: 

59 """CSRF Token管理器 - 生成和验证CSRF tokens""" 

60 

61 def __init__(self): 

62 self.tokens = {} # token存储（生产环境应使用数据库或Redis） 

63 self.token_timeout = 3600 # Token有效期 1小时 

64 

65 def generate_token(self) -> str: 

66 """生成新的CSRF token""" 

67 token = secrets.token_urlsafe(32) 

68 self.tokens[token] = {"created_at": time.time(), "used": False} 

69 return token 

70 

71 def validate_token(self, token: str) -> bool: 

72 """验证CSRF token""" 

73 if not token or token not in self.tokens: 

74 return False 

75 

76 token_info = self.tokens[token] 

77 

78 # 检查是否已使用（防止重放攻击） 

79 if token_info.get("used", False): 

80 return False 

81 

82 # 检查是否过期 

83 if time.time() - token_info["created_at"] > self.token_timeout: 

84 del self.tokens[token] 

85 return False 

86 

87 # 标记为已使用（防止重放攻击） 

88 token_info["used"] = True 

89 return True 

90 

91 def cleanup_expired_tokens(self): 

92 """清理过期的tokens""" 

93 current_time = time.time() 

94 expired_tokens = [ 

95 token for token, info in self.tokens.items() if current_time - info["created_at"] > self.token_timeout 

96 ] 

97 for token in expired_tokens: 

98 del self.tokens[token]

317# SECURITY FIX 1.2: CSRF验证中间件 

318@app.middleware("http") 

319async def csrf_protection_middleware(request: Request, call_next): 

320 """ 

321 CSRF保护中间件 - 验证修改操作的CSRF token 

322 SECURITY: 所有POST/PUT/PATCH/DELETE请求都需要有效的CSRF token 

323 

324 NOTE: 

325 - 在测试环境（testing=True）中自动禁用CSRF保护 

326 - 可通过csrf_enabled配置显式控制（默认True） 

327 - 测试环境会记录调试日志但不阻止请求 

328 """ 

329 from app.core.config import settings 

330 

331 # 确定是否启用CSRF保护 

332 # 测试环境或配置禁用时跳过CSRF验证 

333 should_enforce_csrf = settings.csrf_enabled and not settings.testing 

334 

335 # 对于修改操作，检查CSRF token 

336 if request.method in ["POST", "PUT", "PATCH", "DELETE"]: 

337 if settings.testing: 

338 # 测试环境：记录调试日志但不阻止 

339 logger.debug(f"🧪 CSRF验证跳过 (测试环境): {request.method} {request.url.path}") 

340 elif not settings.csrf_enabled: 

341 # CSRF被显式禁用：记录警告 

342 logger.warning(f"⚠️ CSRF保护已禁用: {request.method} {request.url.path}") 

343 

344 if should_enforce_csrf: 

345 # 某些端点应该排除CSRF检查（如CSRF token生成端点和登录端点） 

346 exclude_paths = [ 

347 "/api/v1/csrf/token", 

348 "/api/csrf-token", 

349 "/api/v1/auth/login", 

350 "/api/v1/auth/register", 

351 "/api/auth/login", # 添加登录端点 

352 "/api/auth/register", # 添加注册端点 

353 "/docs", 

354 "/redoc", 

355 "/openapi.json", 

356 "/swagger-ui", 

357 "/health", # 健康检查 

358 ] 

359 

360 if not any(request.url.path.startswith(path) for path in exclude_paths): 

361 # 获取CSRF token from header 

362 csrf_token = request.headers.get("x-csrf-token") 

363 

364 if not csrf_token: 

365 logger.warning(f"❌ CSRF token missing for {request.method} {request.url.path}") 

366 return JSONResponse( 

367 status_code=403, 

368 content={ 

369 "code": "CSRF_TOKEN_MISSING", 

370 "message": "CSRF token is required for this request", 

371 "data": None, 

372 }, 

373 ) 

374 

375 # 验证CSRF token 

376 if not csrf_manager.validate_token(csrf_token): 

377 logger.warning(f"❌ Invalid CSRF token for {request.method} {request.url.path}") 

378 return JSONResponse( 

379 status_code=403, 

380 content={ 

381 "code": "CSRF_TOKEN_INVALID", 

382 "message": "CSRF token is invalid or expired", 

383 "data": None, 

384 }, 

385 ) 

386 

387 response = await call_next(request) 

388 return response

717def find_available_port(start_port: int, end_port: int) -> int: 

718 """在指定范围内查找可用端口""" 

719 import socket 

720 

721 for port in range(start_port, end_port + 1): 

722 with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as sock: 

723 result = sock.connect_ex(("localhost", port)) 

724 if result != 0: # 端口未被占用 

725 return port 

726 raise RuntimeError(f"No available port found in range {start_port}-{end_port}")
