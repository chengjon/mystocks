23def MA(data: Union[np.ndarray, pd.Series], period: int) -> np.ndarray: 

24 """ 

25 简单移动平均 (Simple Moving Average) 

26 

27 参数: 

28 data: 价格序列 (收盘价、成交量等) 

29 period: 移动平均周期 

30 

31 返回: 

32 np.ndarray: 移动平均值序列 (前period-1个值为NaN) 

33 

34 示例: 

35 >>> close = np.array([10, 11, 12, 13, 14, 15]) 

36 >>> ma5 = MA(close, 5) 

37 >>> print(ma5) 

38 [nan nan nan nan 12. 13.] 

39 """ 

40 if isinstance(data, pd.Series): 

41 data = data.values 

42 

43 if len(data) < period: 

44 return np.full(len(data), np.nan) 

45 

46 # 使用卷积实现移动平均，性能更优 

47 weights = np.ones(period) / period 

48 ma = np.convolve(data, weights, mode="valid") 

49 

50 # 前period-1个值为NaN 

51 result = np.full(len(data), np.nan) 

52 result[period - 1 :] = ma 

53 

54 return result

57def SMA(data: Union[np.ndarray, pd.Series], period: int, weight: float = 1.0) -> np.ndarray: 

58 """ 

59 平滑移动平均 (Smoothed Moving Average) 

60 

61 TDX公式: SMA(X, N, M) = (M*X + (N-M)*Y) / N 

62 其中Y为上一周期SMA值 

63 

64 参数: 

65 data: 价格序列 

66 period: 平滑周期 

67 weight: 权重系数 (默认1.0) 

68 

69 返回: 

70 np.ndarray: 平滑移动平均值序列 

71 

72 示例: 

73 >>> close = np.array([10, 11, 12, 13, 14, 15]) 

74 >>> sma5 = SMA(close, 5, 1) 

75 """ 

76 if isinstance(data, pd.Series): 

77 data = data.values 

78 

79 result = np.zeros_like(data, dtype=float) 

80 result[0] = data[0] 

81 

82 alpha = weight / period 

83 

84 for i in range(1, len(data)): 

85 result[i] = alpha * data[i] + (1 - alpha) * result[i - 1] 

86 

87 # 前period-1个值设为NaN 

88 result[: period - 1] = np.nan 

89 

90 return result

93def EMA(data: Union[np.ndarray, pd.Series], period: int) -> np.ndarray: 

94 """ 

95 指数移动平均 (Exponential Moving Average) 

96 

97 参数: 

98 data: 价格序列 

99 period: EMA周期 

100 

101 返回: 

102 np.ndarray: 指数移动平均值序列 

103 

104 示例: 

105 >>> close = np.array([10, 11, 12, 13, 14, 15]) 

106 >>> ema5 = EMA(close, 5) 

107 """ 

108 if isinstance(data, pd.Series): 

109 data = data.values 

110 

111 alpha = 2.0 / (period + 1) 

112 result = np.zeros_like(data, dtype=float) 

113 result[0] = data[0] 

114 

115 for i in range(1, len(data)): 

116 result[i] = alpha * data[i] + (1 - alpha) * result[i - 1] 

117 

118 # 前period-1个值设为NaN 

119 result[: period - 1] = np.nan 

120 

121 return result

124def CROSS(series1: Union[np.ndarray, pd.Series], series2: Union[np.ndarray, pd.Series]) -> np.ndarray: 

125 """ 

126 交叉函数 - 判断series1是否向上穿越series2 

127 

128 参数: 

129 series1: 第一个序列 (通常是短期均线) 

130 series2: 第二个序列 (通常是长期均线) 

131 

132 返回: 

133 np.ndarray: 布尔数组，True表示发生向上交叉 

134 

135 示例: 

136 >>> ma5 = np.array([10, 11, 12, 13, 14]) 

137 >>> ma10 = np.array([12, 12, 12, 12, 12]) 

138 >>> cross = CROSS(ma5, ma10) 

139 >>> print(cross) 

140 [False False False True False] 

141 """ 

142 if isinstance(series1, pd.Series): 

143 series1 = series1.values 

144 if isinstance(series2, pd.Series): 

145 series2 = series2.values 

146 

147 # 当前周期: series1 > series2 

148 # 前一周期: series1 <= series2 

149 cross = np.zeros(len(series1), dtype=bool) 

150 

151 if len(series1) > 1: 

152 cross[1:] = (series1[1:] > series2[1:]) & (series1[:-1] <= series2[:-1]) 

153 

154 return cross

157def HHV(data: Union[np.ndarray, pd.Series], period: int) -> np.ndarray: 

158 """ 

159 最高值 (Highest High Value) - 滚动窗口内的最高价 

160 

161 参数: 

162 data: 价格序列 

163 period: 滚动窗口周期 

164 

165 返回: 

166 np.ndarray: 滚动最高值序列 

167 

168 示例: 

169 >>> high = np.array([10, 12, 11, 15, 13, 14]) 

170 >>> hhv5 = HHV(high, 5) 

171 >>> print(hhv5) 

172 [nan nan nan nan 15. 15.] 

173 """ 

174 if isinstance(data, pd.Series): 

175 data = data.values 

176 

177 result = np.full(len(data), np.nan) 

178 

179 for i in range(period - 1, len(data)): 

180 result[i] = np.max(data[i - period + 1 : i + 1]) 

181 

182 return result

239def BARSLAST(condition: Union[np.ndarray, pd.Series]) -> np.ndarray: 

240 """ 

241 上一次条件成立位置 - 距离上次条件为True的周期数 

242 

243 参数: 

244 condition: 布尔数组 

245 

246 返回: 

247 np.ndarray: 距离上次True的周期数 

248 

249 示例: 

250 >>> cond = np.array([False, True, False, False, True, False]) 

251 >>> bars = BARSLAST(cond) 

252 >>> print(bars) 

253 [nan 0. 1. 2. 0. 1.] 

254 """ 

255 if isinstance(condition, pd.Series): 

256 condition = condition.values 

257 

258 result = np.full(len(condition), np.nan) 

259 last_true_index = -1 

260 

261 for i, cond_val in enumerate(condition): 

262 if cond_val: 

263 last_true_index = i 

264 result[i] = 0 

265 elif last_true_index >= 0: 

266 result[i] = i - last_true_index 

267 

268 return result

412def MACD( 

413 close: Union[np.ndarray, pd.Series], 

414 fast_period: int = 12, 

415 slow_period: int = 26, 

416 signal_period: int = 9, 

417) -> tuple: 

418 """ 

419 MACD指标 (Moving Average Convergence Divergence) 

420 

421 参数: 

422 close: 收盘价序列 

423 fast_period: 快速EMA周期 (默认12) 

424 slow_period: 慢速EMA周期 (默认26) 

425 signal_period: 信号线周期 (默认9) 

426 

427 返回: 

428 tuple: (DIF, DEA, MACD柱) 

429 

430 示例: 

431 >>> close = np.random.randn(100) + 100 

432 >>> dif, dea, macd = MACD(close) 

433 """ 

434 if isinstance(close, pd.Series): 

435 close = close.values 

436 

437 ema_fast = EMA(close, fast_period) 

438 ema_slow = EMA(close, slow_period) 

439 

440 dif = ema_fast - ema_slow 

441 dea = EMA(dif, signal_period) 

442 macd = (dif - dea) * 2 

443 

444 return dif, dea, macd

447def KDJ( 

448 high: Union[np.ndarray, pd.Series], 

449 low: Union[np.ndarray, pd.Series], 

450 close: Union[np.ndarray, pd.Series], 

451 n: int = 9, 

452 m1: int = 3, 

453 m2: int = 3, 

454) -> tuple: 

455 # pylint: disable=too-many-positional-arguments 

456 

457 """ 

458 

459 

460 KDJ 指标 (随机指标) 

461 

462 参数: 

463 high: 最高价序列 

464 low: 最低价序列 

465 close: 收盘价序列 

466 n: RSV计算周期 (默认9) 

467 m1: K值平滑周期 (默认3) 

468 m2: D值平滑周期 (默认3) 

469 

470 返回: 

471 tuple: (K, D, J) 

472 

473 示例: 

474 >>> k, d, j = KDJ(high, low, close) 

475 """ 

476 if isinstance(high, pd.Series): 

477 high = high.values 

478 if isinstance(low, pd.Series): 

479 low = low.values 

480 if isinstance(close, pd.Series): 

481 close = close.values 

482 

483 # 计算RSV (未成熟随机值) 

484 hhv = HHV(high, n) 

485 llv = LLV(low, n) 

486 

487 rsv = np.where(hhv != llv, (close - llv) / (hhv - llv) * 100, 50) 

488 

489 # 计算K, D, J 

490 k = SMA(rsv, m1, 1) 

491 d = SMA(k, m2, 1) 

492 j = 3 * k - 2 * d 

493 

494 return k, d, j

497def RSI(close: Union[np.ndarray, pd.Series], period: int = 14) -> np.ndarray: 

498 """ 

499 相对强弱指标 (Relative Strength Index) 

500 

501 参数: 

502 close: 收盘价序列 

503 period: RSI周期 (默认14) 

504 

505 返回: 

506 np.ndarray: RSI值序列 (0-100) 

507 

508 示例: 

509 >>> close = np.random.randn(100) + 100 

510 >>> rsi14 = RSI(close, 14) 

511 """ 

512 if isinstance(close, pd.Series): 

513 close = close.values 

514 

515 # 计算价格变化 

516 delta = np.diff(close, prepend=close[0]) 

517 

518 # 分离涨跌 

519 gain = np.where(delta > 0, delta, 0) 

520 loss = np.where(delta < 0, -delta, 0) 

521 

522 # 计算平均涨跌幅 

523 avg_gain = SMA(gain, period, 1) 

524 avg_loss = SMA(loss, period, 1) 

525 

526 # 计算RSI 

527 rs = np.where(avg_loss != 0, avg_gain / avg_loss, 0) 

528 rsi = 100 - (100 / (1 + rs)) 

529 

530 return rsi

533def BOLL(close: Union[np.ndarray, pd.Series], period: int = 20, std_dev: float = 2.0) -> tuple: 

534 """ 

535 布林带指标 (Bollinger Bands) 

536 

537 参数: 

538 close: 收盘价序列 

539 period: 移动平均周期 (默认20) 

540 std_dev: 标准差倍数 (默认2.0) 

541 

542 返回: 

543 tuple: (上轨, 中轨, 下轨) 

544 

545 示例: 

546 >>> upper, middle, lower = BOLL(close, 20, 2) 

547 """ 

548 if isinstance(close, pd.Series): 

549 close = close.values 

550 

551 middle = MA(close, period) 

552 std = STD(close, period) 

553 

554 upper = middle + std_dev * std 

555 lower = middle - std_dev * std 

556 

557 return upper, middle, lower
