63 def save_signals(self, signals: pd.DataFrame, strategy_id: int, batch_insert: bool = True) -> Dict: 

64 """ 

65 保存策略信号到数据库 

66 

67 参数: 

68 signals: 信号DataFrame，必须包含: 

69 - index: DatetimeIndex (信号日期) 

70 - symbol: 股票代码 

71 - signal: 信号类型 ('buy' or 'sell') 

72 - strength: 信号强度 (0.0-1.0，可选) 

73 - entry_price: 入场价格 

74 - indicators: 指标字典 (可选) 

75 - metadata: 元数据字典 (可选) 

76 strategy_id: 策略ID 

77 batch_insert: 是否使用批量插入 

78 

79 返回: 

80 dict: 保存结果 

81 - success: 是否成功 

82 - saved_count: 保存的信号数量 

83 - failed_count: 失败的信号数量 

84 - execution_time: 执行时间(秒) 

85 

86 示例: 

87 >>> signals_df = pd.DataFrame({ 

88 ... 'symbol': ['000001', '000002'], 

89 ... 'signal': ['buy', 'sell'], 

90 ... 'entry_price': [12.34, 56.78], 

91 ... 'strength': [0.8, 0.9] 

92 ... }, index=pd.date_range('2024-01-01', periods=2)) 

93 >>> result = signal_manager.save_signals(signals_df, strategy_id=1) 

94 >>> print(f"保存了 {result['saved_count']} 个信号") 

95 """ 

96 if signals.empty: 

97 self.logger.warning("信号DataFrame为空，无需保存") 

98 return { 

99 "success": True, 

100 "saved_count": 0, 

101 "failed_count": 0, 

102 "execution_time": 0, 

103 } 

104 

105 self.logger.info("开始保存信号: 策略ID=%d, 信号数量=%d", strategy_id, len(signals)) 

106 start_time = datetime.now() 

107 

108 # 验证必需列 

109 required_cols = ["symbol", "signal", "entry_price"] 

110 missing_cols = [col for col in required_cols if col not in signals.columns] 

111 if missing_cols: 

112 raise ValueError(f"信号数据缺少必需列: {missing_cols}") 

113 

114 # 准备数据 

115 signal_records = [] 

116 for idx, row in signals.iterrows(): 

117 # 验证信号类型 

118 if row["signal"] not in ["buy", "sell"]: 

119 self.logger.warning("无效的信号类型: %s, 跳过", row["signal"]) 

120 continue 

121 

122 # 构建记录 

123 record = { 

124 "strategy_id": strategy_id, 

125 "symbol": row["symbol"], 

126 "signal_date": idx.date() if isinstance(idx, pd.Timestamp) else idx, 

127 "signal_type": row["signal"], 

128 "entry_price": float(row["entry_price"]), 

129 "signal_strength": float(row.get("strength", 0.0)), 

130 "indicators": row.get("indicators", {}), 

131 "metadata": row.get("metadata", {}), 

132 "created_at": datetime.now(), 

133 } 

134 

135 # 转换indicators和metadata为JSON字符串 

136 if isinstance(record["indicators"], dict): 

137 record["indicators"] = json.dumps(record["indicators"]) 

138 if isinstance(record["metadata"], dict): 

139 record["metadata"] = json.dumps(record["metadata"]) 

140 

141 signal_records.append(record) 

142 

143 if not signal_records: 

144 self.logger.warning("没有有效的信号记录，保存终止") 

145 return { 

146 "success": False, 

147 "saved_count": 0, 

148 "failed_count": len(signals), 

149 "execution_time": 0, 

150 } 

151 

152 # 批量保存 

153 saved_count = 0 

154 failed_count = 0 

155 

156 try: 

157 if batch_insert and len(signal_records) > self.batch_size: 

158 # 分批保存 

159 for i in range(0, len(signal_records), self.batch_size): 

160 batch = signal_records[i : i + self.batch_size] 

161 batch_df = pd.DataFrame(batch) 

162 

163 try: 

164 self._save_to_database(batch_df) 

165 saved_count += len(batch) 

166 self.logger.debug( 

167 "批次 %d 保存成功: %d 条记录", 

168 i // self.batch_size + 1, 

169 len(batch), 

170 ) 

171 except Exception as e: # pylint: disable=broad-exception-caught 

172 self.logger.error("批次 %d 保存失败: %s", i // self.batch_size + 1, e) 

173 failed_count += len(batch) 

174 else: 

175 # 一次性保存 

176 signals_df = pd.DataFrame(signal_records) 

177 self._save_to_database(signals_df) 

178 saved_count = len(signal_records) 

179 

180 except Exception as e: # pylint: disable=broad-exception-caught 

181 self.logger.error("保存信号时出错: %s", e) 

182 failed_count = len(signal_records) 

183 

184 # 更新统计 

185 execution_time = (datetime.now() - start_time).total_seconds() 

186 self.stats["total_saved"] += saved_count 

187 self.stats["last_save_time"] = datetime.now() 

188 

189 result = { 

190 "success": saved_count > 0, 

191 "saved_count": saved_count, 

192 "failed_count": failed_count, 

193 "execution_time": execution_time, 

194 } 

195 

196 self.logger.info( 

197 "信号保存完成: 成功=%d, 失败=%d, 耗时=%.2f秒", 

198 saved_count, 

199 failed_count, 

200 execution_time, 

201 ) 

202 

203 return result

224 def query_signals( 

225 self, 

226 strategy_id: Optional[int] = None, 

227 symbol: Optional[str] = None, 

228 start_date: Optional[date] = None, 

229 end_date: Optional[date] = None, 

230 signal_type: Optional[str] = None, 

231 limit: int = 1000, 

232 offset: int = 0, 

233 ) -> pd.DataFrame: 

234 # pylint: disable=too-many-positional-arguments 

235 """ 

236 查询策略信号 

237 

238 参数: 

239 strategy_id: 策略ID (可选) 

240 symbol: 股票代码 (可选) 

241 start_date: 开始日期 (可选) 

242 end_date: 结束日期 (可选) 

243 signal_type: 信号类型 ('buy' or 'sell'，可选) 

244 limit: 返回记录数限制 

245 offset: 偏移量 (用于分页) 

246 

247 返回: 

248 pd.DataFrame: 信号记录 

249 

250 示例: 

251 >>> # 查询特定策略的所有买入信号 

252 >>> signals = signal_manager.query_signals( 

253 ... strategy_id=1, 

254 ... signal_type='buy', 

255 ... start_date=date(2024, 1, 1), 

256 ... end_date=date(2024, 12, 31) 

257 ... ) 

258 """ 

259 self.logger.info( 

260 "查询信号: strategy_id=%s, symbol=%s, 日期范围=%s 至 %s, signal_type=%s", 

261 strategy_id, 

262 symbol, 

263 start_date, 

264 end_date, 

265 signal_type, 

266 ) 

267 

268 if self.unified_manager is None: 

269 raise ValueError("UnifiedDataManager未初始化，无法查询信号") 

270 

271 # 构建查询条件 

272 filters = {} 

273 if strategy_id is not None: 

274 filters["strategy_id"] = strategy_id 

275 if symbol is not None: 

276 filters["symbol"] = symbol 

277 if signal_type is not None: 

278 filters["signal_type"] = signal_type 

279 

280 # 日期范围 

281 date_range = {} 

282 if start_date is not None: 

283 date_range["start"] = start_date 

284 if end_date is not None: 

285 date_range["end"] = end_date 

286 

287 # 通过UnifiedDataManager查询 

288 # 注意: 实际实现需要根据UnifiedDataManager的具体接口调整 

289 signals = self.unified_manager.load_data_by_classification( 

290 classification="derived_data", 

291 table_name="strategy_signals", 

292 filters=filters, 

293 date_range=date_range, 

294 limit=limit, 

295 offset=offset, 

296 ) 

297 

298 # 更新统计 

299 self.stats["total_queried"] += len(signals) 

300 self.stats["last_query_time"] = datetime.now() 

301 

302 self.logger.info("查询到 %d 条信号记录", len(signals)) 

303 

304 return signals

368 self, 

369 strategy_id: int, 

370 start_date: Optional[date] = None, 

371 end_date: Optional[date] = None, 

372 ) -> Dict: 

373 """ 

374 获取信号统计信息 

375 

376 参数: 

377 strategy_id: 策略ID 

378 start_date: 开始日期 (可选) 

379 end_date: 结束日期 (可选)

397 start_date=start_date, 

398 end_date=end_date,

400 ) 

401 

402 if signals.empty: 

403 return { 

404 "total_signals": 0, 

405 "buy_signals": 0, 

406 "sell_signals": 0,

411

423 def validate_signal(self, signal: Dict) -> Tuple[bool, Optional[str]]: 

424 """ 

425 验证单个信号的有效性 

426 

427 参数: 

428 signal: 信号字典 

429 

430 返回: 

431 tuple: (是否有效, 错误信息) 

432 

433 示例: 

434 >>> signal = { 

435 ... 'symbol': '000001', 

436 ... 'signal': 'buy', 

437 ... 'entry_price': 12.34, 

438 ... 'strength': 0.8 

439 ... } 

440 >>> is_valid, error = signal_manager.validate_signal(signal) 

441 """ 

442 # 验证必需字段 

443 required_fields = ["symbol", "signal", "entry_price"] 

444 for field in required_fields: 

445 if field not in signal: 

446 return False, f"缺少必需字段: {field}" 

447 

448 # 验证信号类型 

449 if signal["signal"] not in ["buy", "sell"]: 

450 return False, f"无效的信号类型: {signal['signal']}" 

451 

452 # 验证价格 

453 if not isinstance(signal["entry_price"], (int, float)) or signal["entry_price"] <= 0: 

454 return False, f"无效的入场价格: {signal['entry_price']}" 

455 

456 # 验证信号强度 

457 if "strength" in signal: 

458 strength = signal["strength"] 

459 if not isinstance(strength, (int, float)) or not 0 <= strength <= 1: 

460 return False, f"信号强度必须在0-1之间: {strength}" 

461 

462 return True, None
