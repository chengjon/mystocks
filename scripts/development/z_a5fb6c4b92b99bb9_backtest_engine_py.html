36class BacktestEngine: 

37 """ 

38 回测引擎 - 策略回测的统一接口 

39 

40 功能: 

41 - 执行向量化回测 

42 - 计算性能和风险指标 

43 - 生成完整回测报告 

44 - 支持基准比较 

45 """ 

46 

47 def __init__(self, config: Optional[BacktestConfig] = None, risk_free_rate: float = 0.03): 

48 """ 

49 初始化回测引擎 

50 

51 参数: 

52 config: 回测配置 

53 risk_free_rate: 无风险利率（年化） 

54 """ 

55 self.config = config or BacktestConfig() 

56 self.risk_free_rate = risk_free_rate 

57 

58 # 组件 

59 self.backtester = VectorizedBacktester(self.config) 

60 self.perf_metrics = PerformanceMetrics(risk_free_rate) 

61 self.risk_metrics = RiskMetrics() 

62 

63 # 日志 

64 self.logger = logging.getLogger(f"{__name__}.BacktestEngine") 

65 self.logger.setLevel(logging.INFO) 

66 

67 # 结果 

68 self.last_result = None 

69 

70 def run( 

71 self, 

72 price_data: pd.DataFrame, 

73 signals: pd.DataFrame, 

74 benchmark_returns: Optional[pd.Series] = None, 

75 ) -> Dict: 

76 """ 

77 执行回测 

78 

79 参数: 

80 price_data: 价格数据 (open, high, low, close, volume) 

81 signals: 信号数据 (signal, strength) 

82 benchmark_returns: 基准收益率（可选） 

83 

84 返回: 

85 dict: 完整回测结果 

86 - backtest: 回测基础结果 

87 - performance: 性能指标 

88 - risk: 风险指标 

89 - report: 文字报告 

90 

91 示例: 

92 >>> engine = BacktestEngine() 

93 >>> result = engine.run(price_data, signals) 

94 >>> print(result['report']) 

95 """ 

96 self.logger.info("开始执行回测") 

97 

98 # 1. 执行向量化回测 

99 backtest_result = self.backtester.run(price_data, signals) 

100 

101 # 2. 计算性能指标 

102 perf_metrics = self.perf_metrics.calculate_all_metrics( 

103 equity_curve=backtest_result["equity_curve"], 

104 daily_returns=backtest_result["daily_returns"], 

105 trades=backtest_result["trades"], 

106 initial_capital=self.config.initial_capital, 

107 benchmark_returns=benchmark_returns, 

108 ) 

109 

110 # 3. 计算风险指标 

111 risk_metrics_dict = self.risk_metrics.calculate_all_risk_metrics( 

112 equity_curve=backtest_result["equity_curve"], 

113 returns=backtest_result["daily_returns"], 

114 trades=backtest_result["trades"], 

115 total_return=perf_metrics["total_return"], 

116 max_drawdown=perf_metrics["max_drawdown"], 

117 risk_free_rate=self.risk_free_rate, 

118 ) 

119 

120 # 4. 合并所有指标 

121 all_metrics = {**perf_metrics, **risk_metrics_dict} 

122 

123 # 5. 生成报告 

124 perf_report = self.perf_metrics.generate_report(perf_metrics) 

125 risk_report = self.risk_metrics.generate_risk_report(risk_metrics_dict) 

126 full_report = f"{perf_report}\n\n{risk_report}" 

127 

128 # 6. 构建完整结果 

129 result = { 

130 "backtest": backtest_result, 

131 "performance": perf_metrics, 

132 "risk": risk_metrics_dict, 

133 "metrics": all_metrics, 

134 "report": full_report, 

135 "config": self.config, 

136 } 

137 

138 self.last_result = result 

139 self.logger.info("回测执行完成") 

140 

141 return result 

142 

143 def get_trades_df(self) -> pd.DataFrame: 

144 """获取交易记录DataFrame""" 

145 return self.backtester.get_trades_df() 

146

151 return pd.DataFrame() 

152 

153 def get_daily_returns(self) -> pd.Series: 

154 """获取每日收益率""" 

155 if self.last_result: 

156 return self.last_result["backtest"]["daily_returns"] 

157 return pd.Series()

159 def save_result(self, filepath: str): 

160 """ 

161 保存回测结果 

162 

163 参数: 

164 filepath: 保存路径 

165 """ 

166 if not self.last_result: 

167 self.logger.warning("没有可保存的回测结果")

169 

170 # 保存交易记录 

171 trades_df = self.get_trades_df() 

172 if not trades_df.empty: 

173 trades_df.to_csv(f"{filepath}_trades.csv", index=False) 

174 

175 # 保存权益曲线 

176 equity_curve = self.get_equity_curve() 

177 if not equity_curve.empty: 

178 equity_curve.to_csv(f"{filepath}_equity.csv") 

179 

180 # 保存报告 

181 with open(f"{filepath}_report.txt", "w", encoding="utf-8") as f: 

182 f.write(self.last_result["report"])
