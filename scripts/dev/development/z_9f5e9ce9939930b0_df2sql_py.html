1# -*- coding: utf-8 -*- 

2""" 

3Created on Mon May 15 10:01:46 2023 

4@author: CHENGJUN 

5update: 2025-09-12 

6 

7功能介绍：这是用来把dataframe保存到mysql的必备转化工具，可根据dataframe生成SQL建表命令 

8 这个版本是更新过的，它支持设置primary_key和字符集utf8mb4 

9 

10修改日期：2025-09-12 

11修改内容：增加了对大整数的支持，自动选择INT或BIGINT类型 

12""" 

13 

14import pandas as pd 

15 

16 

17def create_sql_cmd(df, table_name, primary_key=None): 

18 # 检查DataFrame是否为空 

19 if df.empty: 

20 raise ValueError("DataFrame 不能为空") 

21 

22 # 获取DataFrame的列名和数据类型 

23 columns = df.dtypes.index.tolist() 

24 dtypes = df.dtypes.values.tolist() 

25 

26 # 创建表的SQL语句，设置默认字符集为utf8mb4以支持中文 

27 create_table_query = f"CREATE TABLE {table_name} (" 

28 

29 # 构建列的定义 

30 for col, dtype in zip(columns, dtypes): 

31 if pd.api.types.is_object_dtype(dtype) or pd.api.types.is_string_dtype(dtype): 

32 max_length = df[col].astype(str).apply(len).max() # 获取列数据的最大长度 

33 varchar_length = min(max_length, 255) # 限制VARCHAR长度最大为255 

34 create_table_query += f"{col} VARCHAR({varchar_length}), " 

35 elif pd.api.types.is_datetime64_any_dtype(dtype): 

36 create_table_query += f"{col} DATETIME, " # 使用DATETIME而不是DATE 

37 elif pd.api.types.is_float_dtype(dtype): 

38 max_decimals = df[col].apply(lambda x: len(str(x).split(".")[-1])).max() # 获取浮点数小数点后的最大位数 

39 float_length = min(max_decimals + 4, 38) # 限制FLOAT小数点后数据长度最大为38 

40 create_table_query += f"{col} FLOAT({float_length}), " 

41 elif pd.api.types.is_integer_dtype(dtype): 

42 # 检查整数的范围，如果超过INT范围则使用BIGINT 

43 max_value = df[col].max() 

44 min_value = df[col].min() 

45 if max_value > 2147483647 or min_value < -2147483648: 

46 create_table_query += f"{col} BIGINT, " 

47 else: 

48 create_table_query += f"{col} INT, " 

49 elif pd.api.types.is_bool_dtype(dtype): 

50 create_table_query += f"{col} TINYINT, " 

51 elif pd.api.types.is_decimal_dtype(dtype): # pylint: disable=no-member 

52 create_table_query += f"{col} DECIMAL(10, 2), " # 假设DECIMAL类型，精度和小数位数可根据需要调整 

53 elif pd.api.types.is_categorical_dtype(dtype): 

54 create_table_query += f"{col} VARCHAR(255), " # 将分类数据视为VARCHAR 

55 else: 

56 create_table_query += f"{col} TEXT, " 

57 

58 # 添加主键定义 

59 if primary_key: 

60 create_table_query += f"PRIMARY KEY ({primary_key}), " 

61 

62 create_table_query = create_table_query.rstrip(", ") + ") CHARACTER SET utf8mb4" 

63 

64 # 返回创建表的SQL语句 

65 return create_table_query
