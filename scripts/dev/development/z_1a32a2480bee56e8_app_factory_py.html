55# SECURITY FIX 1.2: CSRF Token管理 

56class CSRFTokenManager: 

57 """CSRF Token管理器 - 生成和验证CSRF tokens""" 

58 

59 def __init__(self): 

60 self.tokens = {} # token存储（生产环境应使用数据库或Redis） 

61 self.token_timeout = 3600 # Token有效期 1小时 

62 

63 def generate_token(self) -> str: 

64 """生成新的CSRF token""" 

65 token = secrets.token_urlsafe(32) 

66 self.tokens[token] = {"created_at": time.time(), "used": False} 

67 return token 

68 

69 def validate_token(self, token: str) -> bool: 

70 """验证CSRF token""" 

71 if not token or token not in self.tokens: 

72 return False 

73 

74 token_info = self.tokens[token] 

75 

76 # 检查是否已使用（防止重放攻击） 

77 if token_info.get("used", False): 

78 return False 

79 

80 # 检查是否过期 

81 if time.time() - token_info["created_at"] > self.token_timeout: 

82 del self.tokens[token] 

83 return False 

84 

85 # 标记为已使用（防止重放攻击） 

86 token_info["used"] = True 

87 return True 

88 

89 def cleanup_expired_tokens(self): 

90 """清理过期的tokens""" 

91 current_time = time.time() 

92 expired_tokens = [ 

93 token for token, info in self.tokens.items() if current_time - info["created_at"] > self.token_timeout 

94 ] 

95 for token in expired_tokens: 

96 del self.tokens[token] 

97

209 @app.middleware("http") 

210 async def csrf_protection_middleware(request: Request, call_next): 

211 """ 

212 CSRF保护中间件 - 验证修改操作的CSRF token 

213 SECURITY: 所有POST/PUT/PATCH/DELETE请求都需要有效的CSRF token 

214 """ 

215 # OPTIONS请求用于CORS预检，跳过CSRF检查 

216 if request.method == "OPTIONS": 

217 response = await call_next(request) 

218 return response 

219 

220 # 对于修改操作，检查CSRF token 

221 if request.method in ["POST", "PUT", "PATCH", "DELETE"]: 

222 # 某些端点应该排除CSRF检查（如CSRF token生成端点和登录端点） 

223 exclude_paths = [ 

224 "/api/v1/csrf/token", 

225 "/api/csrf-token", 

226 "/api/v1/auth/login", 

227 "/api/v1/auth/register", 

228 "/docs", 

229 "/redoc", 

230 "/openapi.json", 

231 "/swagger-ui", 

232 ] 

233 

234 if not any(request.url.path.startswith(path) for path in exclude_paths): 

235 # 获取CSRF token from header 

236 csrf_token = request.headers.get("x-csrf-token") 

237 

238 if not csrf_token: 

239 logger.warning(f"❌ CSRF token missing for {request.method} {request.url.path}") 

240 request_id = getattr(request.state, "request_id", None) 

241 unified_response = UnifiedResponse( 

242 success=False, 

243 code=BusinessCode.FORBIDDEN, 

244 message="CSRF token is required for this request", 

245 errors=[ 

246 ErrorDetail(code=ErrorCodes.FORBIDDEN, message="CSRF token is required for this request") 

247 ], 

248 request_id=request_id, 

249 ) 

250 return JSONResponse( 

251 status_code=403, 

252 content=unified_response.model_dump(exclude_unset=True), 

253 ) 

254 

255 # 验证CSRF token 

256 if not csrf_manager.validate_token(csrf_token): 

257 logger.warning(f"❌ Invalid CSRF token for {request.method} {request.url.path}") 

258 request_id = getattr(request.state, "request_id", None) 

259 unified_response = UnifiedResponse( 

260 success=False, 

261 code=BusinessCode.FORBIDDEN, 

262 message="CSRF token is invalid or expired", 

263 errors=[ErrorDetail(code=ErrorCodes.FORBIDDEN, message="CSRF token is invalid or expired")], 

264 request_id=request_id, 

265 ) 

266 return JSONResponse( 

267 status_code=403, 

268 content=unified_response.model_dump(exclude_unset=True), 

269 ) 

270 

271 response = await call_next(request) 

272 return response
