60 def _get_cache_key(self, symbol: str, data_type: str, **kwargs) -> str: 

61 """ 

62 生成缓存键 

63 

64 Args: 

65 symbol: 股票代码 

66 data_type: 数据类型 

67 **kwargs: 其他参数 

68 

69 Returns: 

70 str: 缓存键 

71 """ 

72 # 将参数转换为字符串并组合成键 

73 key_parts = [symbol, data_type] 

74 # 按键排序以确保相同参数生成相同的键 

75 for k, v in sorted(kwargs.items()): 

76 key_parts.append(f"{k}={v}") 

77 return "|".join(key_parts)

79 def _get_from_cache(self, cache_key: str): 

80 """ 

81 从缓存中获取数据 

82 

83 Args: 

84 cache_key: 缓存键 

85 

86 Returns: 

87 缓存的数据，如果不存在或过期则返回None 

88 """ 

89 if cache_key in self.data_cache: 

90 cached_item = self.data_cache[cache_key] 

91 # 检查缓存是否过期（这里设置为5分钟） 

92 if (datetime.now() - cached_item["timestamp"]).total_seconds() < 300: 

93 logger.info("使用缓存数据: %s", cache_key) 

94 return cached_item["data"] 

95 else: 

96 # 删除过期缓存 

97 del self.data_cache[cache_key] 

98 return None
