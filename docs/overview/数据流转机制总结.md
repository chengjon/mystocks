## 🎯 核心概念

### 数据分类架构 (US3 - Unified Storage Strategy 3)

系统采用**5层数据分类**，共34个分类，每类数据自动路由到最优数据库：

| 分类层级 | 数据类型 | 目标数据库 | 示例分类数 |
|---------|---------|-----------|-----------|
| **第1类** | 市场数据 (高频) | TDengine | 6项 |
| **第2类** | 参考数据 (低频) | PostgreSQL | 9项 |
| **第3类** | 衍生数据 (计算) | PostgreSQL | 6项 |
| **第4类** | 交易数据 (事务) | PostgreSQL | 7项 |
| **第5类** | 元数据 (配置) | PostgreSQL | 6项 |

#### 3.1 统一保存接口 (`src/core/data_manager.py`)

```python
class DataManager:
    def save_data(
        self,
        classification: DataClassification,
        data: pd.DataFrame,
        table_name: str,
        **kwargs
    ) -> bool:
        """
        保存数据到正确的数据库

        Args:
            classification: 数据分类
            data: DataFrame数据
            table_name: 目标表名
            **kwargs: 额外参数 (upsert, conflict_columns等)

        Returns:
            bool: 保存是否成功
        """
        # 1. 快速路由决策
        target_db = self.get_target_database(classification)

        # 2. 根据目标数据库选择访问层
        if target_db == DatabaseTarget.TDENGINE:
            success = self._tdengine.save_data(
                data, classification, table_name, **kwargs
            )
        else:  # PostgreSQL
            success = self._postgresql.save_data(
                data, classification, table_name, **kwargs
            )

        # 3. 记录性能指标
        if self.enable_monitoring:
            self._performance_monitor.record_operation(
                operation="save_data",
                classification=classification.value,
                duration_ms=duration_ms,
                success=success
            )

        return success
```

#### 3.2 PostgreSQL数据访问层 (`src/data_access/postgresql_access.py`)

**批量插入 (Insert)**:
```python
def insert_dataframe(self, table_name: str, df: pd.DataFrame) -> int:
    """
    批量插入DataFrame (使用execute_values优化)

    性能优化:
    - execute_values: 一次插入多行
    - 批处理: 显著减少网络往返
    - 事务管理: 确保数据一致性

    Returns:
        插入的行数
    """
    if df.empty:
        return 0

    conn = self._get_connection()

    try:
        # 1. 准备列名和数据
        columns = list(df.columns)
        columns_str = ", ".join(columns)
        data = [tuple(row) for row in df.itertuples(index=False, name=None)]

        # 2. 构建插入SQL
        sql = f"INSERT INTO {table_name} ({columns_str}) VALUES %s"

        # 3. 批量执行 (execute_values优化)
        cursor = conn.cursor()
        execute_values(cursor, sql, data)
        conn.commit()

        rows_inserted = cursor.rowcount or 0
        return rows_inserted
    except Exception as e:
        conn.rollback()
        raise
    finally:
        self._return_connection(conn)
```

**批量更新 (Upsert)**:
```python
def upsert_dataframe(
    self,
    table_name: str,
    df: pd.DataFrame,
    conflict_columns: List[str],
    update_columns: Optional[List[str]] = None
) -> int:
    """
    批量Upsert (INSERT ... ON CONFLICT UPDATE)

    使用场景:
    - 数据去重
    - 增量更新
    - 解决主键冲突

    Example:
        upsert_dataframe('daily_kline', df,
                       conflict_columns=['symbol', 'date'],
                       update_columns=['close', 'volume'])
    """
    conn = self._get_connection()

    try:
        columns = list(df.columns)
        data = [tuple(row) for row in df.itertuples(index=False, name=None)]

        # 1. 构建冲突处理子句
        conflict_str = ", ".join(conflict_columns)
        if update_columns is None:
            update_columns = [col for col in columns if col not in conflict_columns]

        update_str = ", ".join([f"{col} = EXCLUDED.{col}" for col in update_columns])

        # 2. Upsert SQL
        sql = f"""
            INSERT INTO {table_name} ({columns_str})
            VALUES %s
            ON CONFLICT ({conflict_str})
            DO UPDATE SET {update_str}
        """

        # 3. 执行
        cursor = conn.cursor()
        execute_values(cursor, sql, data)
        conn.commit()

        return cursor.rowcount or 0
    except Exception as e:
        conn.rollback()
        raise
    finally:
        self._return_connection(conn)
```

#### 3.3 连接池管理

```python
class PostgreSQLDataAccess:
    """PostgreSQL连接池管理"""

    def _get_connection(self):
        """从连接池获取连接"""
        if self.pool is None:
            self.pool = self.conn_manager.get_postgresql_connection()
        return self.pool.getconn()

    def _return_connection(self, conn):
        """归还连接到连接池"""
        if self.pool:
            self.pool.putconn(conn)
```

#### 4.2 时间范围查询 (PostgreSQL)

```python
def query_by_time_range(
    self,
    table_name: str,
    time_column: str,
    start_time: str,
    end_time: str,
    symbol: Optional[str] = None,
    limit: Optional[int] = None
) -> pd.DataFrame:
    """
    时间范围查询 (使用TimescaleDB优化)

    性能优化:
    - 时间索引
    - 分区裁剪
    - 倒序查询优化
    """
    conn = self._get_connection()

    try:
        # 构建查询SQL
        where_conditions = [
            f"{time_column} >= '{start_time}'",
            f"{time_column} <= '{end_time}'"
        ]

        if symbol:
            where_conditions.append(f"symbol = '{symbol}'")

        where_clause = " AND ".join(where_conditions)
        order_clause = f"ORDER BY {time_column} DESC"
        limit_clause = f"LIMIT {limit}" if limit else ""

        sql = f"""
            SELECT * FROM {table_name}
            WHERE {where_clause}
            {order_clause}
            {limit_clause}
        """

        df = pd.read_sql(sql, conn)
        return df
    finally:
        self._return_connection(conn)
```

#### 5.2 认证机制

```python
# JWT Token生成
def create_access_token(data: Dict[str, Any]) -> str:
    """
    创建JWT访问令牌

    配置:
        - Algorithm: HS256
        - Secret: JWT_SECRET_KEY (from .env)
        - Expires: 30 minutes
    """
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(minutes=30)
    to_encode.update({"exp": expire})
    encoded_jwt = jwt.encode(to_encode, settings.secret_key, algorithm=settings.algorithm)
    return encoded_jwt

# 认证依赖
async def get_current_user(token: str = Depends(oauth2_scheme)) -> User:
    """
    验证JWT令牌并返回当前用户

    流程:
        1. 提取令牌
        2. 验证签名
        3. 解码载荷
        4. 查询用户信息
        5. 返回用户对象
    """
    try:
        payload = jwt.decode(token, settings.secret_key, algorithms=[settings.algorithm])
        username: str = payload.get("sub")
        user_id: int = payload.get("user_id")
        role: str = payload.get("role")

        if username is None:
            raise HTTPException(status_code=401, detail="Could not validate credentials")

        token_data = TokenData(username=username, user_id=user_id, role=role)
    except jwt.PyJWTError:
        raise HTTPException(status_code=401, detail="Could not validate credentials")

    # 从数据库获取用户信息
    user = authenticate_user_by_id(token_data.user_id)
    if user is None:
        raise HTTPException(status_code=401, detail="User not found")

    return user
```
