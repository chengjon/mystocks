22def validate_identifier(identifier: str, identifier_type: str = "identifier") -> str: 

23 """ 

24 验证和清洗SQL标识符（表名、列名等）以防止SQL注入 

25 

26 TDengine/PostgreSQL标识符规则: 

27 - 只能包含字母、数字、下划线 

28 - 必须以字母或下划线开头 

29 - 不包含SQL特殊字符 

30 

31 Args: 

32 identifier: 待验证的标识符 

33 identifier_type: 标识符类型（用于错误消息） 

34 

35 Returns: 

36 验证后的安全标识符 

37 

38 Raises: 

39 ValueError: 如果标识符包含不安全字符 

40 

41 Examples: 

42 >>> validate_identifier("my_table", "table_name") 

43 'my_table' 

44 >>> validate_identifier("'; DROP TABLE users; --", "table_name") 

45 ValueError: Invalid table_name: ''; DROP TABLE users; --' 

46 """ 

47 if not identifier: 

48 raise ValueError(f"{identifier_type} cannot be empty") 

49 

50 # 检查是否只包含安全字符 

51 if not re.match(r"^[a-zA-Z_][a-zA-Z0-9_]*$", identifier): 

52 raise ValueError( 

53 f"Invalid {identifier_type}: '{identifier}'. " f"Only alphanumeric characters and underscores are allowed." 

54 ) 

55 

56 return identifier

91def validate_symbol(symbol: str) -> str: 

92 """ 

93 验证股票代码格式 

94 

95 Args: 

96 symbol: 股票代码（如 "000001.SZ" 或 "000001"） 

97 

98 Returns: 

99 验证后的安全股票代码 

100 

101 Raises: 

102 ValueError: 如果股票代码格式无效 

103 

104 Examples: 

105 >>> validate_symbol("000001") 

106 '000001' 

107 >>> validate_symbol("000001.SZ") 

108 '000001.SZ' 

109 >>> validate_symbol("'; DROP TABLE; --") 

110 ValueError: Invalid symbol format 

111 """ 

112 if not symbol: 

113 raise ValueError("Symbol cannot be empty") 

114 

115 # 清理：移除不安全字符，只保留字母数字和点 

116 clean_symbol = re.sub(r"[^a-zA-Z0-9.]", "", symbol) 

117 

118 if not clean_symbol: 

119 raise ValueError(f"Invalid symbol format: '{symbol}'") 

120 

121 return clean_symbol

124def escape_string_value(value: Any) -> str: 

125 """ 

126 转义字符串值中的单引号以防止SQL注入 

127 

128 Args: 

129 value: 待转义的值 

130 

131 Returns: 

132 转义后的安全字符串（带单引号） 

133 

134 Examples: 

135 >>> escape_string_value("O'Brien") 

136 "'O''Brien'" 

137 >>> escape_string_value(None) 

138 "NULL" 

139 """ 

140 if value is None: 

141 return "NULL" 

142 

143 if isinstance(value, str): 

144 # 转义单引号（SQL标准方式：单引号加倍） 

145 escaped = value.replace("'", "''") 

146 return f"'{escaped}'" 

147 

148 # 对于数字类型，直接转换为字符串 

149 if isinstance(value, (int, float, bool)): 

150 return str(value) 

151 

152 # 其他类型转换为字符串后转义 

153 escaped = str(value).replace("'", "''") 

154 return f"'{escaped}'"

157def build_safe_insert_sql( 

158 table_name: str, 

159 columns: List[str], 

160 values: List[Tuple[Any, ...]], 

161 using_clause: Optional[str] = None, 

162 tags_clause: Optional[str] = None, 

163) -> str: 

164 """ 

165 构建安全的INSERT语句（防止SQL注入） 

166 

167 Args: 

168 table_name: 表名 

169 columns: 列名列表（会被验证） 

170 values: 值元组列表（会被转义） 

171 using_clause: TDengine USING子句（可选） 

172 tags_clause: TDengine TAGS子句（可选） 

173 

174 Returns: 

175 安全的INSERT SQL语句 

176 

177 Raises: 

178 ValueError: 如果表名或列名不安全 

179 

180 Examples: 

181 >>> columns = ["ts", "price", "volume"] 

182 >>> values = [("2024-01-01 10:00:00", 10.5, 1000)] 

183 >>> build_safe_insert_sql("my_table", columns, values) 

184 "INSERT INTO my_table (ts, price, volume) VALUES ('2024-01-01 10:00:00', 10.5, 1000)" 

185 """ 

186 # 验证表名 

187 safe_table_name = validate_table_name(table_name) 

188 

189 # 验证列名 

190 safe_columns = [validate_column_name(col) for col in columns] 

191 

192 # 构建列列表 

193 columns_str = ", ".join(safe_columns) 

194 

195 # 构建值列表（转义每个值） 

196 value_groups = [] 

197 for value_tuple in values: 

198 escaped_values = [escape_string_value(v) for v in value_tuple] 

199 value_groups.append(f"({', '.join(escaped_values)})") 

200 

201 values_str = ", ".join(value_groups) 

202 

203 # 构建SQL 

204 sql = f"INSERT INTO {safe_table_name} ({columns_str}) VALUES {values_str}" 

205 

206 # 添加USING和TAGS子句（TDengine） 

207 if using_clause: 

208 safe_using = validate_table_name(using_clause) 

209 sql = f"INSERT INTO {safe_table_name} USING {safe_using}" 

210 elif tags_clause: 

211 sql = f"INSERT INTO {safe_table_name}" 

212 

213 if tags_clause: 

214 sql += f" {tags_clause}" 

215 

216 if using_clause or tags_clause: 

217 sql += f" VALUES {values_str}" 

218 

219 return sql

222def build_safe_select_sql( 

223 table_name: str, where_conditions: Optional[dict] = None, columns: Optional[List[str]] = None 

224) -> str: 

225 """ 

226 构建安全的SELECT语句（防止SQL注入） 

227 

228 Args: 

229 table_name: 表名 

230 where_conditions: WHERE条件字典 {列名: 值} 

231 columns: 要选择的列名列表（None表示SELECT *） 

232 

233 Returns: 

234 安全的SELECT SQL语句 

235 

236 Raises: 

237 ValueError: 如果表名或列名不安全 

238 

239 Examples: 

240 >>> build_safe_select_sql("my_table", {"symbol": "000001", "date": "2024-01-01"}) 

241 "SELECT * FROM my_table WHERE symbol = '000001' AND date = '2024-01-01'" 

242 """ 

243 # 验证表名 

244 safe_table_name = validate_table_name(table_name) 

245 

246 # 构建列列表 

247 if columns: 

248 safe_columns = [validate_column_name(col) for col in columns] 

249 columns_str = ", ".join(safe_columns) 

250 else: 

251 columns_str = "*" 

252 

253 # 构建WHERE条件 

254 where_clauses = [] 

255 if where_conditions: 

256 for col, value in where_conditions.items(): 

257 safe_col = validate_column_name(col) 

258 escaped_value = escape_string_value(value) 

259 where_clauses.append(f"{safe_col} = {escaped_value}") 

260 

261 where_str = " WHERE " + " AND ".join(where_clauses) if where_clauses else "" 

262 

263 # 构建SQL 

264 sql = f"SELECT {columns_str} FROM {safe_table_name}{where_str}" 

265 

266 return sql
