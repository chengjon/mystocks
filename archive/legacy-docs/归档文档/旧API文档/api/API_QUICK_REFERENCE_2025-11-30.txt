🔑 认证端点
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1️⃣  获取 CSRF Token
   方法: GET
   路径: /api/csrf-token
   描述: 获取 CSRF Token (可选)

   示例:
   $ curl -X GET "http://localhost:8020/api/csrf-token"

2️⃣  用户登录
   方法: POST
   路径: /api/auth/login
   描述: 用户登录，返回 JWT Token

   请求体:
   {
     "username": "admin",
     "password": "admin123"
   }

   示例:
   $ curl -X POST "http://localhost:8020/api/auth/login" \
     -H "Content-Type: application/json" \
     -d '{"username":"admin","password":"admin123"}'

   响应:
   {
     "access_token": "eyJ...",
     "token_type": "bearer",
     "user": {...}
   }

3️⃣  获取用户信息
   方法: GET
   路径: /api/auth/user
   描述: 获取当前登录用户信息 (需要 Token)

   请求头:
   Authorization: Bearer <access_token>

   示例:
   $ curl -X GET "http://localhost:8020/api/auth/user" \
     -H "Authorization: Bearer <jwt-token>"

📝 Python 示例
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

import requests

# 登录
response = requests.post(
    "http://localhost:8020/api/auth/login",
    json={"username": "admin", "password": "admin123"}
)
data = response.json()
token = data["access_token"]

# 获取用户信息
headers = {"Authorization": f"Bearer {token}"}
user_response = requests.get(
    "http://localhost:8020/api/auth/user",
    headers=headers
)
print(user_response.json())


📋 JavaScript 示例
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

// 登录
const loginRes = await fetch('http://localhost:8020/api/auth/login', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    username: 'admin',
    password: 'admin123'
  })
});

const { access_token } = await loginRes.json();

// 获取用户信息
const userRes = await fetch('http://localhost:8020/api/auth/user', {
  headers: { 'Authorization': `Bearer ${access_token}` }
});

const user = await userRes.json();
console.log(user);

💡 常用 cURL 命令
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

curl -s http://localhost:8020/health | python3 -m json.tool

# 获取 Token
TOKEN=$(curl -s -X POST "http://localhost:8020/api/auth/login" \
  -H "Content-Type: application/json" \
  -d '{"username":"admin","password":"admin123"}' | grep -o '"access_token":"[^"]*' | cut -d'"' -f4)

# 使用 Token 请求数据
curl -s -H "Authorization: Bearer $TOKEN" \
  http://localhost:8020/api/auth/user | python3 -m json.tool
